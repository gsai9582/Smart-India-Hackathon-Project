function updateDisplay() {
  document.getElementById("player1-score").innerText = gameState.player1.score;
  document.getElementById("player1-score-detail").innerText = gameState.player1.score;
  document.getElementById("player1-strikes").innerText = gameState.player1.strikes;
  document.getElementById("player1-misses").innerText = gameState.player1.misses;
  document.getElementById("player1-accuracy").innerText = gameState.player1.accuracy;
  document.getElementById("player1-progress").style.width = gameState.player1.accuracy + "%";

  document.getElementById("player2-score").innerText = gameState.player2.score;
  document.getElementById("player2-score-detail").innerText = gameState.player2.score;
  document.getElementById("player2-strikes").innerText = gameState.player2.strikes;
  document.getElementById("player2-misses").innerText = gameState.player2.misses;
  document.getElementById("player2-accuracy").innerText = gameState.player2.accuracy;
  document.getElementById("player2-progress").style.width = gameState.player2.accuracy + "%";
}

function updateAllCharts() {
  gameState.updateCount++;
  const maxHistory = 10;
  
  // Score history
  gameState.scoreHistory.player1.push(gameState.player1.score);
  gameState.scoreHistory.player2.push(gameState.player2.score);
  if (gameState.scoreHistory.player1.length > maxHistory) {
    gameState.scoreHistory.player1.shift();
    gameState.scoreHistory.player2.shift();
  }
  
  // Accuracy history
  gameState.accuracyHistory.player1.push(gameState.player1.accuracy);
  gameState.accuracyHistory.player2.push(gameState.player2.accuracy);
  if (gameState.accuracyHistory.player1.length > maxHistory) {
    gameState.accuracyHistory.player1.shift();
    gameState.accuracyHistory.player2.shift();
  }
  
  // Efficiency history
  const p1Efficiency = Math.round((gameState.player1.strikes / (gameState.player1.strikes + gameState.player1.misses)) * 100);
  const p2Efficiency = Math.round((gameState.player2.strikes / (gameState.player2.strikes + gameState.player2.misses)) * 100);
  gameState.efficiencyHistory.player1.push(p1Efficiency);
  gameState.efficiencyHistory.player2.push(p2Efficiency);
  if (gameState.efficiencyHistory.player1.length > maxHistory) {
    gameState.efficiencyHistory.player1.shift();
    gameState.efficiencyHistory.player2.shift();
  }
  
  const timeLabels = gameState.scoreHistory.player1.map((_, i) => 'T' + (i+1));

  // Update existing charts
  gameState.charts.scoreChart.data.labels = timeLabels;
  gameState.charts.scoreChart.data.datasets[0].data = gameState.scoreHistory.player1;
  gameState.charts.scoreChart.data.datasets[1].data = gameState.scoreHistory.player2;
  gameState.charts.scoreChart.update('none');

  gameState.charts.accuracyChart.data.datasets[0].data = 
    gameState.accuracyHistory.player1.map((acc, i) => ({x: i+1, y: acc}));
  gameState.charts.accuracyChart.data.datasets[1].data = 
    gameState.accuracyHistory.player2.map((acc, i) => ({x: i+1, y: acc}));
  gameState.charts.accuracyChart.update('none');

  gameState.charts.efficiencyChart.data.labels = timeLabels;
  gameState.charts.efficiencyChart.data.datasets[0].data = gameState.efficiencyHistory.player1;
  gameState.charts.efficiencyChart.data.datasets[1].data = gameState.efficiencyHistory.player2;
  gameState.charts.efficiencyChart.update('none');

  // Update individual player strikes charts
  gameState.charts.alexStrikesChart.data.datasets[0].data = [gameState.player1.strikes];
  gameState.charts.alexStrikesChart.data.datasets[1].data = [gameState.player1.misses];
  gameState.charts.alexStrikesChart.update('none');

  gameState.charts.sarahStrikesChart.data.datasets[0].data = [gameState.player2.strikes];
  gameState.charts.sarahStrikesChart.data.datasets[1].data = [gameState.player2.misses];
  gameState.charts.sarahStrikesChart.update('none');
}

function simulateLiveMatch() {
  if (Math.random() > 0.25) {
    gameState.player1.score += Math.floor(Math.random() * 3) + 1;
    gameState.player1.strikes += Math.floor(Math.random() * 4) + 1;
  }
  if (Math.random() > 0.75) gameState.player1.misses++;

  if (Math.random() > 0.35) {
    gameState.player2.score += Math.floor(Math.random() * 2) + 1;
    gameState.player2.strikes += Math.floor(Math.random() * 3) + 1;
  }
  if (Math.random() > 0.65) gameState.player2.misses++;

  const p1Total = gameState.player1.strikes + gameState.player1.misses;
  const p2Total = gameState.player2.strikes + gameState.player2.misses;
  gameState.player1.accuracy = p1Total > 0 ? Math.round((gameState.player1.strikes / p1Total) * 100) : 88;
  gameState.player2.accuracy = p2Total > 0 ? Math.round((gameState.player2.strikes / p2Total) * 100) : 76;

  updateDisplay();
  updateAllCharts();
}

// Canvas animation
const canvas = document.getElementById('courtCanvas');
const ctx2 = canvas.getContext('2d');
function resizeCanvas() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
resizeCanvas(); window.addEventListener('resize', resizeCanvas);

const balls = [{x:100,y:100,dx:3,dy:2,radius:10,color:'#ff4500'}, {x:200,y:150,dx:2,dy:3,radius:10,color:'#00aaff'}];
const bats = [{x:300,y:200,dx:2.5,dy:1.5,width:60,height:10,color:'#964B00'}, {x:500,y:300,dx:-3,dy:2,width:60,height:10,color:'#8B0000'}];

function drawBall(ball) {
  ctx2.beginPath(); ctx2.arc(ball.x, ball.y, ball.radius, 0, Math.PI*2); ctx2.fillStyle=ball.color; ctx2.fill(); ctx2.closePath();
}
function drawBat(bat) {
  ctx2.fillStyle=bat.color; ctx2.fillRect(bat.x,bat.y,bat.width,bat.height); ctx2.fillRect(bat.x+bat.width/2-3,bat.y+bat.height,6,15);
}
function animate() {
  ctx2.clearRect(0,0,canvas.width,canvas.height);
  balls.forEach(ball => { ball.x+=ball.dx; ball.y+=ball.dy; if(ball.x+ball.radius>canvas.width||ball.x-ball.radius<0)ball.dx*=-1; if(ball.y+ball.radius>canvas.height||ball.y-ball.radius<0)ball.dy*=-1; drawBall(ball); });
  bats.forEach(bat => { bat.x+=bat.dx; bat.y+=bat.dy; if(bat.x<0||bat.x+bat.width>canvas.width)bat.dx*=-1; if(bat.y<0||bat.y+bat.height+15>canvas.height)bat.dy*=-1; drawBat(bat); });
  requestAnimationFrame(animate);
}
animate();
